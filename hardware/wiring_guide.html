Communication Protocol

Data Exchange Format

Raspberry Pi ‚Üí Arduino (Commands)

SPEED:50,STEER:20\n

Format: "SPEED:[value],STEER:[value]"

Example: "SPEED:75,STEER:-15" - Move forward at 75% speed, turn left 15 degrees

Arduino ‚Üí Raspberry Pi (Sensor Data)

DIST:25.5,LIR:120,RIR:80\n

Format: "DIST:[cm],LIR:[value],RIR:[value]"

Example: "DIST:30.2,LIR:150,RIR:45" - Object 30.2cm away, left IR: 150, right IR: 45

Arduino Code

// Motor control pins const int motorLeft = 5; const int motorRight = 6; const int steeringServo = 3; // Sensor pins const int trigPin = 11; const int echoPin = 12; const int irLeft = A0; const int irRight = A1; void setup() { Serial.begin(9600); // Initialize pins... } void loop() { // Read sensors float distance = readUltrasonic(); int leftIR = analogRead(irLeft); int rightIR = analogRead(irRight); // Send sensor data to Raspberry Pi Serial.print("DIST:"); Serial.print(distance); Serial.print(",LIR:"); Serial.print(leftIR); Serial.print(",RIR:"); Serial.println(rightIR); // Check for commands from Raspberry Pi if (Serial.available() > 0) { String command = Serial.readStringUntil('\n'); processCommand(command); } delay(50); } void processCommand(String command) { // Parse commands like "SPEED:50,STEER:20" int speed = 0; int steering = 0; if (command.startsWith("SPEED:")) { int commaIndex = command.indexOf(','); if (commaIndex != -1) { speed = command.substring(6, commaIndex).toInt(); steering = command.substring(commaIndex + 7).toInt(); } } // Execute commands setMotorSpeed(speed); setSteering(steering); }
